### 소수 찾기

```java
import java.util.*;

class Solution {
    
    static Set<Integer> numSet = new HashSet<Integer>();
    static boolean[] checked;
        
    public int solution(String numbers) {
        checked = new boolean[numbers.length()];
        
        for (int i=1; i<numbers.length() +1; i++) {
            generate(numbers, "", i);
        }
        
        int answer = 0;
        
        for (int num : numSet) {
            if (isPrimeNumber(num)) {
                answer++;
            }
        }
        
        return answer;
    }
    
    public void generate(String numbers, String generated, int len) {
        
        if (len == generated.length()) {
            numSet.add(Integer.parseInt(generated));
            return;
        }
        
        for (int i = 0; i<numbers.length() ; i++) {
            if (!checked[i]) {
                generated += numbers.charAt(i);
                checked[i] = true;

                generate(numbers, generated, len);
                checked[i] = false;
                generated = generated.substring(0, generated.length()-1);
            }
        }
    }
    
    public boolean isPrimeNumber(int num) {
        if ( (num == 0) || (num ==1) ) {
            return false;
        }
        
        for (int i = 2; i <= Math.sqrt(num); i++) {
            if (num%i == 0) {
                return false;
            }
        }
        
        return true;
    }
    
}
```
